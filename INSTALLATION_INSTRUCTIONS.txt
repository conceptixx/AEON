NEXUS v2 Enterprise - Installation Instructions
=============================================================

TARGET LOCATION: /aeon/runtime/python/nexus

BACKUP LOCATION: /aeon/runtime/python/nexus_v2_backup_YYYYMMDD

=============================================================
INSTALLATION STEPS
=============================================================

1. BACKUP CURRENT INSTALLATION
   -------------------------------------------------------------
   cd /aeon/runtime/python
   cp -r nexus nexus_v2_backup_$(date +%Y%m%d)

2. EXTRACT THIS ZIP
   -------------------------------------------------------------
   cd /tmp
   unzip nexus_v2_enterprise_complete.zip
   cd nexus_v2_enterprise_complete

3. REPLACE/ADD FILES
   -------------------------------------------------------------
   cd /aeon/runtime/python/nexus

   # Replace updated files
   cp /tmp/nexus_v2_enterprise_complete/core/loader.py core/
   cp /tmp/nexus_v2_enterprise_complete/core/module.py core/
   cp /tmp/nexus_v2_enterprise_complete/daemon.py .

   # Add new directories
   cp -r /tmp/nexus_v2_enterprise_complete/aeon_interface/ .
   cp -r /tmp/nexus_v2_enterprise_complete/docs/ .

   # Create module structure
   mkdir -p modules/mesh
   mkdir -p modules/cortex
   mkdir -p modules/autonomic
   mkdir -p modules/substrate
   touch modules/mesh/__init__.py
   touch modules/cortex/__init__.py
   touch modules/autonomic/__init__.py
   touch modules/substrate/__init__.py

4. SETUP AEON INTERFACE
   -------------------------------------------------------------
   mkdir -p /aeon/interfaces/python/nexus
   cd /aeon/interfaces/python/nexus

   ln -sf /aeon/runtime/python/nexus/aeon_interface/__init__.py __init__.py
   ln -sf /aeon/runtime/python/nexus/aeon_interface/status.py status.py
   ln -sf /aeon/runtime/python/nexus/aeon_interface/live.py live.py
   ln -sf /aeon/runtime/python/nexus/aeon_interface/command.py command.py

5. TEST INSTALLATION
   -------------------------------------------------------------
   # Test basic daemon
   python /aeon/runtime/python/nexus/example_daemon.py

   # Test AEON interface
   python /aeon/interfaces/python/nexus/live.py liveness
   python /aeon/interfaces/python/nexus/status.py

6. (OPTIONAL) INSTALL ENTERPRISE FEATURES
   -------------------------------------------------------------
   # Install as needed
   pip install nexus-v2-redis-state      # Distributed state
   pip install nexus-v2-cluster           # HA clustering
   pip install nexus-v2-tracing           # Distributed tracing
   pip install nexus-v2-multitenancy      # Multi-tenancy
   pip install nexus-v2-compliance        # GDPR/SOX/HIPAA
   pip install nexus-v2-georeplication    # Multi-DC
   pip install nexus-v2-slo               # SLO tracking
   pip install nexus-v2-chaos             # Chaos engineering
   pip install nexus-v2-finops            # Cost management
   pip install nexus-v2-bluegreen         # Blue-green deploys
   pip install nexus-v2-mtls              # Mutual TLS
   pip install nexus-v2-loadbalancer      # Load balancing
